CREATE TABLE "JBS_LOGIN_AUDIT" (	
	"ID" NUMBER NOT NULL, 
	"TIME" VARCHAR2(25 CHAR) NOT NULL, 
	"DBA_USER" VARCHAR2(20 CHAR) NOT NULL, 
	"USER_ID" NUMBER NOT NULL, 
	CONSTRAINT "JBS_LOGIN_AUDIT_PK" PRIMARY KEY ("ID"), 
	CONSTRAINT "JBS_LOGIN_AUDIT_FK1" FOREIGN KEY ("USER_ID")
	REFERENCES "JBS_USER" ("ID") ON DELETE CASCADE
);
